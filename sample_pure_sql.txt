SELECT client.ID client_id, client.NAME client_name, (SELECT SUM(task.PRICE) FROM px_task task WHERE task.CLIENT_ID = client.ID AND task.STATUS_ID = (SELECT px_task_status.ID FROM px_task_status WHERE CODE = 'F')) total_price_F, (SELECT SUM(task.PRICE) FROM px_task task WHERE task.CLIENT_ID = client.ID AND task.STATUS_ID = (SELECT px_task_status.ID FROM px_task_status WHERE CODE = 'P')) total_price_P, (SELECT COUNT(task.ID) FROM px_task task WHERE task.CLIENT_ID = client.ID) as total_tasks FROM px_client client LEFT JOIN px_task task ON task.CLIENT_ID = client.ID GROUP BY client.ID
